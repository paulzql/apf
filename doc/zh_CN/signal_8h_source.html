214  // bind between signal and slot
215  template<class T>
216  bool Bind(T* obj, void (T::*func)(T1)) {
217  bool ret = true;
218  lock_mutex(&mutex_);
219  for(unsigned int i = 0; i < slot_set_.size(); i++) {
220  if (slot_set_[i]->equals(obj, reinterpret_cast<void*>(&func))) {
221  ret = false;
222  break;
223  }
224  }
225  if (ret) {
226  slot_set_.push_back(new Slot<T1>(obj, func));
227  }
228  unlock_mutex(&mutex_);
229 
230  return ret;
231  }

239  // unbind between signal and slot
240  template<class T>
241  void UnBind(T* obj, void (T::*func)(T1)) {
242  lock_mutex(&mutex_);
243  for(unsigned int i = 0; i < slot_set_.size(); i++) {
244  if (slot_set_[i]->equals(obj, reinterpret_cast<void*>(&func))) {
245  delete slot_set_[i];
246  // move front
247  for (i++; i < slot_set_.size(); i++) {
248  slot_set_[i-1] = slot_set_[i];
249  }
250  slot_set_.resize(slot_set_.size()-1);
251  break;
252  }
253  }
254  unlock_mutex(&mutex_);
255  }
