55 bool PluginManager::Load(const char* path) {
56  APF_DEBUG("loading module %s\n", path);
57  if (modules_.end() != modules_.find(path)) {
58  APF_DEBUG("load module failed (alread loaded)\n");
59  return false;
60  }
61 
62  HMODULE hmodule = NULL;
63  ClassEntry* classes = NULL;
64  unsigned long version = 0;
65  APF_GET_MODULE_INFO module_func = NULL;
66  APF_SET_OBJECT_CREATOR set_object_creator_func = NULL;
67  #ifdef WIN32
68  hmodule = ::LoadLibraryExA(path, NULL, LOAD_WITH_ALTERED_SEARCH_PATH);
69  if (NULL != hmodule) {
70  module_func = (APF_GET_MODULE_INFO)GetProcAddress(hmodule, APF_GET_MODULE_INFO_NAME);
71  set_object_creator_func = (APF_SET_OBJECT_CREATOR)GetProcAddress(hmodule, APF_SET_OBJECT_CREATOR_NAME);
72  }
73  #else
74  hmodule = dlopen(path, RTLD_LAZY);
75  if (NULL != hmodule) {
76  module_func = (APF_GET_MODULE_INFO)dlsym(hmodule, APF_GET_MODULE_INFO_NAME);
77  set_object_creator_func = (APF_SET_OBJECT_CREATOR)dlsym(hmodule, APF_SET_OBJECT_CREATOR_NAME);
78  }
79  #endif
80  if (NULL == hmodule) {
81  APF_DEBUG("open module file failed\n");
82  return false;
83  }
84  if (NULL == module_func) {
85  APF_DEBUG("can't find module interface : %s\n", APF_GET_MODULE_INFO_NAME);
86  return false;
87  }
88  // get module info
89  version = module_func(APF_VERSION(major_version_, sub_version_), &classes);
90  if (0 == version) {
91  APF_DEBUG("incompatible module\n");
92  return false;
93  }
94  if (NULL == classes) {
95  APF_DEBUG("no classes found\n");
96  return false;
97  }
98 
99  if (NULL != set_object_creator_func) {
100  set_object_creator_func(apf::APFCreateObject);
101  }
102 
103  RegisterClasses(classes);
104 
105  ModuleItem item;
106  item.hmodule = hmodule;
107  item.classes = classes;
108  item.version = version;
109 
110  modules_.insert(std::map<std::string, ModuleItem>::value_type(path, item));
111 
112  APF_DEBUG("loaded module %s\n", path);
113 
114  return true;
115 }

117 bool PluginManager::UnLoad(const char* path) {
118  std::map<std::string, ModuleItem>::iterator it = modules_.find(path);
119  if (modules_.end() == modules_.find(path)) {
120  APF_DEBUG("no module found %s\n", path);
121  return false;
122  }
123 
124  UnRegisterClasses(it->second.classes);
125  #ifdef WIN32
126  FreeLibrary(it->second.hmodule);
127  #else
128  dlclose(it->second.hmodule);
129  #endif
130  modules_.erase(it);
131 
132  return true;
133 }

149 void PluginManager::RegisterClasses(const ClassEntry* classes) {
150  const ClassEntry empty_class;
151  while (classes && (!classes->equals(empty_class))) {
152  if (!Class::RegisterClass(*classes)) {
153  // register class failed
154  APF_DEBUG("Register class failed [classid:%s,name:%s]\n", classes->clsid.c_str(), classes->class_name);
155  } else {
156  APF_DEBUG("Register class ok [classid:%s,name:%s]\n", classes->clsid.c_str(), classes->class_name);
157  }
158  classes++;
159  }
160 }
