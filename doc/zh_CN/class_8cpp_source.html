32  bool ret = false;

34  lock_mutex(&global_mutex);

38  ret = true;
39  } else if (replace) {
40  class_map_[class_entry.clsid] = class_entry;
41  ret = true;
42  }
43  unlock_mutex(&global_mutex);

46 }
