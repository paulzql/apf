179 int wait_semaphore(sem_t *psem, unsigned long waittime) {
180  struct timespec ts;
181 
182  #ifdef __APPLE__
183  /*
184  // OS X does not have clock_gettime, use clock_get_time
185  clock_serv_t cclock;
186  mach_timespec_t mts;
187  host_get_clock_service(mach_host_self(), CALENDAR_CLOCK, &cclock);
188  clock_get_time(cclock, &mts);
189  mach_port_deallocate(mach_task_self(), cclock);
190  ts.tv_sec = mts.tv_sec;
191  ts.tv_nsec = mts.tv_nsec;
192  */
193  int ret = sem_trywait(psem);
194  while ((waittime > 0) && ((0 != ret) && (EINVAL != errno))) {
195  unsigned long sleep_time = (waittime < 1000 ? waittime : 1000);
196  msleep(sleep_time);
197  waittime -= sleep_time;
198  ret = sem_trywait(psem);
199  }
200  return ret;
201  #else
202  clock_gettime(CLOCK_REALTIME, &ts);
203  #endif
204 
205  ts.tv_sec = time(NULL) + waittime /1000;
206  ts.tv_nsec += (waittime % 1000 ) * 1000;
207 
208 #ifdef __APPLE__
209  /*
210  int sem_timedwait (sem_t *sem, const struct timespec *abs_timeout);
211  return sem_timedwait((sem_t*)(*psem), &ts);
212  */
213 #else
214  return sem_timedwait(psem, &ts);
215 #endif
216 }

236  {
237  init = 1;
238 
239  uint64_t tick;

241  #if defined(WIN32)

243  __asm
244  {
245  rdtsc
246  mov lowtick, eax
247  mov hightick, edx
248  }
249  tick = hightick;
250  tick <<= 32;
251  tick |= lowtick;
252  #elif defined(__GNUC__) && ( defined(__i686__) || defined(__i386__) )
253  asm("rdtsc" : "=A" (tick));
254  #elif defined (__SUNPRO_CC) || defined( __sparc__ )
255  tick = gethrtime();
256  #elif defined(__MACH__) || defined(__linux)
257  int fd=open("/dev/random",O_RDONLY);
258  read(fd,&tick,sizeof(tick));
259  closesocket(fd);
260  #else
261  # error Need some way to seed the random number generator
262  #endif

264  #ifdef WIN32
265  srand(seed);
266  #else
267  srandom(seed);
268  #endif
269  }

273  int r2 = rand();
274

276

280  #endif
